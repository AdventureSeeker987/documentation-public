title: "useWalletConnector"

Get wallet connection.

- [Tech docs](/sdks/tech-doc/modules/orderly_network_hooks#usewalletconnector)

```ts
export interface WalletConnectorContextState {
  connect: () => Promise<WalletState[]>;
  disconnect: (options: any) => Promise<any[]>;
  connecting: boolean;
  setChain: (options: { chainId: string | number }) => Promise<any>;
  chains: any[];
  wallet: any;
  connectedChain: ConnectedChain | null;
  settingChain: boolean;
}
```

### Example

```tsx
const wallet = useWalletConnector();

return (
  <button
    onClick={() => {
      wallet.connect();
    }}
  >
    Connect wallet
  </button>
);
```
