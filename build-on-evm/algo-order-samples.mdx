## Create Algo Order Sample


### STOP
`STOP` order will be triggered once the given `trigger_price` is reached and new order of `trigger_price_type` will be created.
```json
{
    "symbol": "PERP_NEAR_USDC", 
    "algo_type": "STOP", 
    "quantity": "5.5", 
    "side": "BUY", 
    "type": "LIMIT", 
    "trigger_price_type": "MARK_PRICE", 
    "trigger_price": "4.203", 
    "price": "3.5"
}
```

### TP_SL

```json
{
    "symbol": "PERP_NEAR_USDC",
    "algo_type": "TP_SL",
    "quantity": "5.5",
    "trigger_price_type": "MARK_PRICE",
    "order_tag": "test_tag",
    "child_orders": [
        {
            "symbol": "PERP_NEAR_USDC",
            "algo_type": "TAKE_PROFIT",
            "side": "SELL",
            "type": "MARKET",
            "trigger_price": 3.365,
            "reduce_only": true
        },
        {
            "symbol": "PERP_NEAR_USDC",
            "algo_type": "STOP_LOSS",
            "side": "SELL",
            "type": "MARKET",
            "trigger_price": 3.36,
            "reduce_only": true
        }
    ]
}
```

### POSTIONAL_TP_SL
```json
{
    "symbol": "PERP_NEAR_USDC", 
    "algo_type": "POSITIONAL_TP_SL", 
    "trigger_price_type": "MARK_PRICE", 
    "child_orders": [
        {
            "symbol": "PERP_NEAR_USDC", 
            "algo_type": "TAKE_PROFIT", 
            "side": "SELL", 
            "type": "CLOSE_POSITION", 
            "trigger_price_type": "MARK_PRICE", 
            "trigger_price": 4.05, 
            "reduce_only": true
        }, 
        {
            "symbol": "PERP_NEAR_USDC", 
            "algo_type": "STOP_LOSS", 
            "side": "SELL", 
            "type": "CLOSE_POSITION", 
            "trigger_price_type": "MARK_PRICE", 
            "trigger_price": 3.95, 
            "reduce_only": true
        }
    ]
}
```
